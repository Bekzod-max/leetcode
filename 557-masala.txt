function reverseWords(s: string): string {
  let result: string = '';
  let s1: string[] = s.split(' ');

  for (let i = 0; i < s1.length; i++) {
    let b = '';
    for (let j = 0; j < s1[i].length ; j++) {
      b =  s1[i][j] + b;
    }
    result += b + ' ';
  }

  return result.trim();
};
