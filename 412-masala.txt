function fizzBuzz(n: number): string[]
{
  let res : string[] = [];
  for (let i = 1; i <= n; i++)
  {
    if( i % 3 != 0 && i % 5 != 0)
    {
      res.push(i.toString());
    }
    else  if (i % 3 === 0 && i % 5 == 0)
    {
      res.push('FizzBuzz');
    }
    else  if (i % 3 === 0)
    {
      res.push('Fizz');
    }
    else if (i % 5 === 0)
    {
      res.push('Buzz');
    }

  }
  return res;
};
